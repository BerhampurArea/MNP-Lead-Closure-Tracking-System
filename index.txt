const pinIndex = {};
PIN_MAP.forEach(r => { pinIndex[r.pin] = r; });

const pinInput = document.getElementById('pincode');
const assignedBox = document.getElementById('assignedBox');
const assignedJP = document.getElementById('assignedJP');
const assignedJC = document.getElementById('assignedJC');

let currentAssignment = null;

pinInput.addEventListener('input', () => {
  const val = pinInput.value.trim();
  if (val.length === 6 && pinIndex[val]) {
    currentAssignment = pinIndex[val];
    assignedJP.textContent = currentAssignment.jpid;
    assignedJC.textContent = currentAssignment.jc;
    assignedBox.classList.add('show');
  } else {
    currentAssignment = null;
    assignedBox.classList.remove('show');
  }
});

document.getElementById('leadForm').addEventListener('submit', async function(e) {
  e.preventDefault();
  const submitBtn = document.getElementById('submitBtn');
  const statusMsg = document.getElementById('statusMsg');

  const pincode = pinInput.value.trim();
  if (!pinIndex[pincode]) {
    statusMsg.className = 'status-msg error';
    statusMsg.textContent = 'Pin code not recognized in Berhampur Area mapping. Please verify the pin code.';
    return;
  }

  const payload = {
    timestamp: new Date().toISOString(),
    prmId: document.getElementById('prmId').value.trim(),
    custName: document.getElementById('custName').value.trim(),
    mobile: document.getElementById('mobile').value.trim(),
    address: document.getElementById('address').value.trim(),
    block: document.getElementById('block').value.trim(),
    district: document.getElementById('district').value.trim(),
    pincode: pincode,
    assignedJpId: currentAssignment.jpid,
    assignedJC: currentAssignment.jc,
    status: "New"
  };

  submitBtn.disabled = true;
  submitBtn.textContent = "Submitting...";
  statusMsg.className = 'status-msg';
  statusMsg.style.display = 'none';

  try {
    await fetch(SCRIPT_URL, {
      method: 'POST',
      mode: 'no-cors',
      headers: { 'Content-Type': 'text/plain' },
      body: JSON.stringify(payload)
    });
    statusMsg.className = 'status-msg success';
    statusMsg.textContent = 'Lead submitted successfully! Assigned to ' + currentAssignment.jpid + '.';
    document.getElementById('leadForm').reset();
    assignedBox.classList.remove('show');
  } catch (err) {
    statusMsg.className = 'status-msg error';
    statusMsg.textContent = 'Submission failed. Please check your connection and try again.';
  } finally {
    submitBtn.disabled = false;
    submitBtn.textContent = "Submit Lead";
  }
});
